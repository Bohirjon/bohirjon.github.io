// Segregated interfaces based on specific roles
public interface IWorkable
{
    void Work();
}

public interface IEatable
{
    void Eat();
}

public interface IProgrammable
{
    void Program();
}

public interface IPlumber
{
    void FixPlumbing();
}
// Classes implement only relevant interfaces
public class Developer : IWorkable, IEatable, IProgrammable
{
    public void Work() => Console.WriteLine("Writing code");
    public void Eat() => Console.WriteLine("Eating lunch");
    public void Program() => Console.WriteLine("Programming");
}

public class Plumber : IWorkable, IEatable, IPlumber
{
    public void Work() => Console.WriteLine("Fixing pipes");
    public void Eat() => Console.WriteLine("Eating lunch");
    public void FixPlumbing() => Console.WriteLine("Plumbing work");
}

public class Robot : IWorkable, IProgrammable
{
    public void Work() => Console.WriteLine("Performing tasks");
    public void Program() => Console.WriteLine("Executing code");
    // Robot doesn't eat - no need to implement IEatable!
}
