{

        string couponCode = null, 
        int loyaltyPoints = 0)
    {
        if (!string.IsNullOrEmpty(couponCode))
        {
            ApplyCoupon(couponCode); // future feature
        }

        if (loyaltyPoints > 0)

            ApplyLoyaltyPoints(loyaltyPoints); // speculative
        }

        Console.WriteLine($"Order processed for {amount}");
    }
